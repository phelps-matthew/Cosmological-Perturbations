\noindent For $\Omega(t)$,
and  
in gauge $P=1/2$, $J=0$, $R=-1$
\[
	\eta^{\alpha\beta} \pd_\alpha h_{\nu\beta} = \frac12\Omega^2 \pd_\nu h - \Omega h \pd_\nu \Omega
\]
\ba
	\delta G_{00} 	&=  \frac12 \Box(\Omega^{-2}h_{00}) -\frac12 \Omega^{-4}\pd_0(h_{00}\pd_0(\Omega^2))+6\Omega^{-4}h_{00}(\pd_0 \Omega)^2-\frac14 \Box h + \frac14 \Omega^2h\Box (\Omega^{-2}) \\
	\delta G_{ij} &=  \frac12 \Box(\Omega^{-2} h_{ij})-\frac12 \Omega^{-4}\pd_0(h_{ij}\pd_0(\Omega^2))-3\Omega^{-3}h_{ij}\Box\Omega + 2\Omega^{-4}h_{ij}(\pd_0 \Omega)^2-\frac14 \Omega^{-2}\Box(\Omega^2 h)-\delta_{ij}\Omega^{-1}h_{00}\Box(\Omega^{-1})\\
	\delta G_{0i} &=  \frac12 \Box(\Omega^{-2} h_{0i})-\frac12 \Omega^{-4}\pd_0(h_{0i}\pd_0(\Omega^2))-2\Omega^{-3}h_{0i}\Box \Omega + 3 \Omega^{-4}h_{0i}(\pd_0 \Omega)^2-\frac14 h \Omega^{2}\Box(\Omega^{-2})
\ea

\pd_0\pd_i h
\ea
